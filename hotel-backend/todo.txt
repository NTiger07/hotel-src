router.post("/add", async (req, res) => {
  try {
    const { room_number, start_date, end_date } = req.body;
    
    // Check if the room exists
    const room = await Room.findOne({ room_number });
    if (!room) {
      res.status(404);
      return res.send("Room not found");
    }

    // Check if the room is already booked for the given time range
    const existingBooking = await Booking.findOne({
      room_number,
      $or: [
        { start_date: { $lte: end_date }, end_date: { $gte: start_date } },
        { start_date: { $gte: start_date, $lte: end_date } }
      ]
    });
    
    if (existingBooking) {
      res.status(400);
      return res.send("Room already booked for the given time range");
    }

    // If the room is available, create the booking
    const newBooking = await Booking.create(req.body);
    await room.updateOne({ status: "booked" });
    
    res.status(200).send("Booking Added");
  } catch (error) {
    console.error(error);
    res.status(500).send("Internal Server Error");
  }
});
